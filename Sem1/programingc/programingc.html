Logical Operators
#include <stdio.h>
int main() {
// Logical Operators
int h = 0, j = 1;

// Logical AND operator (&&)
printf("The logical AND operator returned %d\n", h && j);

// Logical OR operator (||)
printf("The logical OR operator returned %d\n", h || j);

// Logical NOT operator (!)
printf("The logical NOT operator on j returned %d\n", !j);
printf("The logical NOT operator on h returned %d\n", !h);

return 0;
}

Output

The logical AND operator returned 0
The logical OR operator returned 1
The logical NOT operator on j returned 0
The logical NOT operator on h returned 1

Bitwise Operators
// Bitwise Operators
// A = 00111100
// B = 00001101
// R = 00001100
#include <stdio.h>
int main() {
int A = 60, B = 14;
// Bitwise AND operator
printf("Bitwise AND operator returned %d\n", A & B);
// Bitwise OR operator
printf("Bitwise OR operator returned %d\n", A | B);
// Bitwise XOR operator
printf("Bitwise XOR operator returned %d\n", A ^ B);
// Bitwise one's complement operator
printf("Bitwise one's complement operator returned %d\n", ~B);
// Bitwise left shift operator
printf("Bitwise left shift operator returned %d\n", A << 2);
// Bitwise right shift operator
printf("Bitwise right shift operator returned %d\n", A >> 2);
return 0;
}
Output:
Bitwise AND operator returned 12
Bitwise OR operator returned 62
Bitwise XOR operator returned 50
Bitwise one's complement operator returned -15
Bitwise left shift operator returned 240
Bitwise right shift operator returned 15

The Additional, Subtractionand And Multiplication of Two Numbers
#include <stdio.h>
int main() {
printf("Enter First Number: ");
int num1;
int num2;
scanf("%d", &num2); // Take input for num2
printf("Enter Second Number: ");
scanf("%d", &num1); // Take input for num1
printf("The additional number of two is %d\n", num1 + num2); // Sum of num1 and num2
printf("The subtraction number of two is %d\n", num1 - num2); // Difference of num1 and num2
printf("The multiplication of two number is %d\n", num1 * num2); // Product of num1 and num2
return 0;
}
Output:
Enter First Number: 9
Enter Second Number: 5
The additional number of two is 14
The subtraction number of two is 4
The multiplication of two number is 45

Student Marks Calculation , Percentage and Pass or Fail
#include <stdio.h>
int main() {
int math, science, english, computer, total;
float percentage;
printf("Enter marks for 4 subjects (out of 100):\n");
printf("Math: ");
scanf("%d", &math);
printf("Science: ");
scanf("%d", &science);
printf("English: ");
scanf("%d", &english);
printf("Computer: ");
scanf("%d", &computer);
total = math + science + english + computer;
percentage = total / 4.0;
if (math < 40 || science < 40 || english < 40 || computer < 40) {
printf("You have failed in one or more subjects.\n");
} else if (percentage >= 40) {
if (percentage >= 40 && percentage < 50) {
printf("You have to attend the chance exam. Your score is %.2f%%.\n", percentage);
} else {
printf("You have passed. You scored %.2f%%.\n", percentage);
}
} else {
printf("You have failed. Your score is %.2f%%.\n", percentage);
}
return 0;
}
Example Output:
Enter marks for 4 subjects (out of 100):
Math: 50
Science: 60
English: 45
Computer: 70
You have passed. You scored 56.25%.
